063  public static Tweet readTweet(JsonNode json, List<String> contentKeys, List<String> featureKeys, boolean handleEntities) {
064    if ( (contentKeys == null) || (featureKeys == null) ) {
065      return new Tweet(json, handleEntities);
066    }
067
068    // implied else
069    return new Tweet(json, contentKeys, featureKeys, handleEntities);
070  }

073  /**
074   * Used by the JSONTWeetFormat; the DocumentContent contains only the main text;
075   * the annotation feature map contains all the other JSON data, recursively.
076   */
077  private Tweet(JsonNode json, boolean handleEntities) {
078    string = "";
079    Iterator<String> keys = json.fieldNames();
080    FeatureMap features = Factory.newFeatureMap();
081    annotations = new HashSet<PreAnnotation>();
082
083    while (keys.hasNext()) {
084      String key = keys.next();
085
086      if (key.equals(TweetUtils.DEFAULT_TEXT_ATTRIBUTE)) {
087        RepositioningInfo repos = new RepositioningInfo();
088        string = unescape(json.get(key).asText(), repos);
089        if(handleEntities) processEntities(json, 0L, repos);
090      } else if(key.equals("entities") && handleEntities) {
091        // do nothing - don't add entities as a feature
092      } else {
093        features.put(key.toString(), TweetUtils.process(json.get(key)));
094      }
095    }
096    
097    annotations.add(new PreAnnotation(0L, string.length(), TweetUtils.TWEET_ANNOTATION_TYPE, features));
098  }

101  /** Used by the fancier corpus population system to handle options.
102   * @param contentKeys JSON paths whose values should be converted to String and
103   * added to the DocumentContent
104   * @param featureKeys JSON paths whose values should be stored in the main
105   * annotation's features
106   */
107  private Tweet(JsonNode json, List<String> contentKeys, List<String> featureKeys, boolean handleEntities) {
108    StringBuilder content = new StringBuilder();
109    List<String> keepers = new ArrayList<String>();
110    keepers.addAll(contentKeys);
111    keepers.addAll(featureKeys);
112    this.annotations = new HashSet<PreAnnotation>();
113
114    FeatureMap featuresFound = TweetUtils.process(json, keepers);
115
116    // Put the DocumentContent together from the contentKeys' values found in the JSON.
117    for (String cKey : contentKeys) {
118      if (featuresFound.containsKey(cKey)) {
119        int start = content.length();
120        // Use GATE's String conversion in case there are maps or lists.
121        String str = Strings.toString(featuresFound.get(cKey));
122        RepositioningInfo repos = null;
123        if(TweetUtils.DEFAULT_TEXT_ATTRIBUTE.equals(cKey)) {
124          repos = new RepositioningInfo();
125          str = unescape(str, repos);
126        }
127        content.append(str);
128        this.annotations.add(new PreAnnotation(start, content.length(), cKey));
129        if(handleEntities && TweetUtils.DEFAULT_TEXT_ATTRIBUTE.equals(cKey)) {
130          // only process entities within "text"
131          processEntities(json, start, repos);
132        }
133        content.append('\n');
134      }
135    }
136    
137    // Get the featureKeys & their values for the main annotation.
138    FeatureMap annoFeatures = Factory.newFeatureMap();
139    for (String fKey : featureKeys) {
140      if (featuresFound.containsKey(fKey)) {
141        annoFeatures.put(fKey, featuresFound.get(fKey));
142      }
143    }
144    
145    // Create the main annotation and the content.
146    this.annotations.add(new PreAnnotation(0, content.length(), TweetUtils.TWEET_ANNOTATION_TYPE, annoFeatures));
147    this.string = content.toString();
148  }

150  private static Pattern XML_ENTITY_PATTERN = Pattern.compile("&(amp|lt|gt);");
151  
152  /**
153   * Un-escape &amp;amp;, &amp;gt; and &amp;lt; in the given string, populating
154   * the supplied {@link RepositioningInfo} to describe the offset changes.
155   * @param str string, possibly including escaped ampersands or angle brackets
156   * @param repos {@link RepositioningInfo} to hold offset changes
157   * @return the unescaped string
158   */
159  private String unescape(String str, RepositioningInfo repos) {
160    StringBuffer buf = new StringBuffer();
161    int correction = 0;
162    int lastMatchEnd = 0;
163    Matcher mat = XML_ENTITY_PATTERN.matcher(str);
164    while(mat.find()) {
165      if(mat.start() != lastMatchEnd) {
166        // repositioning record for the span from end of previous match to start of this one
167        int nonMatchLen = mat.start() - lastMatchEnd;
168        repos.addPositionInfo(lastMatchEnd, nonMatchLen, lastMatchEnd - correction, nonMatchLen);
169      }
170      // repositioning record covering this match
171      repos.addPositionInfo(mat.start(), mat.end() - mat.start(), mat.start() - correction, 1);
172      correction += mat.end() - mat.start() - 1;
173      String replace = "?";
174      switch(mat.group(1)) {
175        case "amp": replace = "&"; break;
176        case "gt": replace = ">"; break;
177        case "lt": replace = "<"; break;
178      }
179      mat.appendReplacement(buf, replace);
180      lastMatchEnd = mat.end();
181    }
182    int tailLen = str.length() - lastMatchEnd;
183    if(tailLen > 0) {
184      // repositioning record covering everything after the last match
185      repos.addPositionInfo(lastMatchEnd, tailLen, lastMatchEnd - correction, tailLen);
186    }
187    mat.appendTail(buf);
188    return buf.toString();
189  }

191  /**
192   * Process the "entities" property of this json object into annotations,
193   * shifting their offsets by the specified amount.
194   * 
195   * @param json the Tweet json object
196   * @param startOffset offset correction if the text is not the first of
197   *         the content keys.
198   */
199  private void processEntities(JsonNode json, long startOffset, RepositioningInfo repos) {
200    JsonNode entitiesNode = json.get(TweetUtils.ENTITIES_ATTRIBUTE);
201    if(entitiesNode == null || !entitiesNode.isObject()) {
202      // no entities, nothing to do
203      return;
204    }
205    Iterator<String> entityTypes = entitiesNode.fieldNames();
206    while(entityTypes.hasNext()) {
207      String entityType = entityTypes.next();
208      JsonNode entitiesOfType = entitiesNode.get(entityType);
209      if(entitiesOfType != null && entitiesOfType.isArray() && entitiesOfType.size() > 0) {
210        // if the entityType is X:Y then assume X is the AS name and Y is the actual type
211        String[] setAndType = entityType.split(":", 2);
212        Iterator<JsonNode> it = entitiesOfType.elements();
213        while(it.hasNext()) {
214          JsonNode entity = it.next();
215          if(entity.isObject()) {
216            // process is guaranteed to return a FeatureMap given an object
217            FeatureMap features = (FeatureMap)TweetUtils.process(entity);
218            Object indices = features.get("indices");
219            if(indices != null && indices instanceof List<?>) {
220              List<?> indicesList = (List<?>)indices;
221              if(indicesList.get(0) instanceof Number && indicesList.get(1) instanceof Number) {
222                // finally we know we have a valid entity
223                features.remove("indices");
224                long annStart = repos.getExtractedPos(startOffset + ((Number)indicesList.get(0)).longValue());
225                long annEnd = repos.getExtractedPos(startOffset + ((Number)indicesList.get(1)).longValue());
226                if(setAndType.length == 2) {
227                  // explicit annotation set name
228                  annotations.add(new PreAnnotation(annStart, annEnd, setAndType[0], setAndType[1], features));
229                } else {
230                  annotations.add(new PreAnnotation(annStart, annEnd, entityType, features));
231                }
232              }
233            }
234          }
235        }
236      }
237    }
238  }
