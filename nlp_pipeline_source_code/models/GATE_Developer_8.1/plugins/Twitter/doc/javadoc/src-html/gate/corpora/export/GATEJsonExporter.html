048    comment = "Export documents and corpora in JSON format",
049    tool = true, autoinstances = @AutoInstance, icon = "GATEJSON")

119  public GATEJsonExporter() {
120    super("GATE JSON", "json","application/json");
121  }
122
123  @Override
124  public void export(Document doc, OutputStream out, FeatureMap options)
125    throws IOException {
126    try(JsonGenerator generator = openGenerator(out, options)) {
127      export(doc, generator, options);
128    }
129  }
130  
131  public void export(Corpus corpus, OutputStream out, FeatureMap options)
132    throws IOException {
133    try(JsonGenerator generator = openGenerator(out, options)) {
134      Iterator<Document> docIter = corpus.iterator();
135      int currentDocIndex = 0;
136      while(docIter.hasNext()) {
137        boolean docWasLoaded =
138                corpus.isDocumentLoaded(currentDocIndex);
139        Document currentDoc = docIter.next();
140        try {
141          export(currentDoc, generator, options);
142        } finally {
143          // unload if necessary
144          if(!docWasLoaded) {
145            corpus.unloadDocument(currentDoc);
146            Factory.deleteResource(currentDoc);
147          }
148          currentDocIndex++;
149        }
150      }
151    }
152  }
153  
154  /**
155   * Create a JsonGenerator ready to write to the given output stream.
156   * If the specified options indicate that we want to wrap the output
157   * in an array then output the array start event in preparation.
158   */
159  protected JsonGenerator openGenerator(OutputStream out, FeatureMap options)
160    throws IOException {
161    JsonGenerator generator = MAPPER.getFactory().createGenerator(out);
162    generator.disable(JsonGenerator.Feature.AUTO_CLOSE_TARGET);
163    generator.enable(JsonGenerator.Feature.AUTO_CLOSE_JSON_CONTENT);
164    if(options.containsKey("exportAsArray") && ((Boolean)options.get("exportAsArray")).booleanValue()) {
165      generator.writeStartArray();
166    } else {
167      // writing concatenated, put newlines in between
168      generator.setRootValueSeparator(new SerializedString("\n"));
169    }
170    
171    return generator;
172  }
173
174  public void export(Document doc, JsonGenerator generator, FeatureMap options)
175    throws IOException {
176    try {
177      AnnotationSet defaultEntitiesAS =
178        doc.getAnnotations((String)options.get("entitiesAnnotationSetName"));
179      
180      @SuppressWarnings("unchecked")
181      Collection<String> types = (Collection<String>)options.get("annotationTypes");
182      
183      Map<String,Collection<Annotation>> annotationsMap = new LinkedHashMap<>();
184      
185      for (String type : types) {
186        String[] setAndType = type.split(":", 2);
187        if(setAndType.length == 1) {
188          annotationsMap.put(type, defaultEntitiesAS.get(type));
189        } else {
190          annotationsMap.put(type, doc.getAnnotations(setAndType[0]).get(setAndType[1]));
191        }
192      }
193      
194      // look for document annotations
195      AnnotationSet docAnnots = null;
196      if(options.containsKey("documentAnnotationType")) {
197        docAnnots = doc.getAnnotations((String)options.get("documentAnnotationASName"))
198                .get((String)options.get("documentAnnotationType"));
199      }
200      if(docAnnots == null || docAnnots.isEmpty()) {
201        // no document annotations, write everything
202        Map<String, Collection<Annotation>> sortedAnnots = new LinkedHashMap<>();
203        for(Map.Entry<String, Collection<Annotation>> entry : annotationsMap.entrySet()) {
204          sortedAnnots.put(entry.getKey(), Utils.inDocumentOrder((AnnotationSet)entry.getValue()));
205        }
206        DocumentJsonUtils.writeDocument(doc, 0L, Utils.end(doc), sortedAnnots, null, null, generator);
207      } else {
208        for(Annotation docAnnot : Utils.inDocumentOrder(docAnnots)) {
209          Map<String, Collection<Annotation>> coveredAnnotations = new HashMap<>();
210          for(Map.Entry<String, Collection<Annotation>> entry : annotationsMap.entrySet()) {
211            coveredAnnotations.put(entry.getKey(),
212                    Utils.inDocumentOrder(((AnnotationSet)entry.getValue()).getContained(
213                            Utils.start(docAnnot), Utils.end(docAnnot))));
214          }
215          DocumentJsonUtils.writeDocument(doc, Utils.start(docAnnot), Utils.end(docAnnot),
216                  coveredAnnotations, docAnnot.getFeatures(), null, generator);
217        }
218      }
219    } catch(InvalidOffsetException e) {
220      // should never happen, as all offsets come from the document itself
221      throw new LuckyException("Invalid offset found within document", e);
222    }
223  }
